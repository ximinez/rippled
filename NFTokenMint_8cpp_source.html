35 NFTokenMint::preflight(PreflightContext const& ctx)
36 {
37  if (!ctx.rules.enabled(featureNonFungibleTokensV1))
38  return temDISABLED;
39 
40  if (auto const ret = preflight1(ctx); !isTesSuccess(ret))
41  return ret;
42 
43  if (ctx.tx.getFlags() & tfNFTokenMintMask)
44  return temINVALID_FLAG;
45 
46  if (auto const f = ctx.tx[~sfTransferFee])
47  {
48  if (f > maxTransferFee)
49  return temBAD_NFTOKEN_TRANSFER_FEE;
50 
51  // If a non-zero TransferFee is set then the tfTransferable flag
52  // must also be set.
53  if (f > 0u && !ctx.tx.isFlag(tfTransferable))
54  return temMALFORMED;
55  }
56 
57  // An issuer must only be set if the tx is executed by the minter
58  if (auto iss = ctx.tx[~sfIssuer]; iss == ctx.tx[sfAccount])
59  return temMALFORMED;
60 
61  if (auto uri = ctx.tx[~sfURI])
62  {
63  if (uri->length() == 0 || uri->length() > maxTokenURILength)
64  return temMALFORMED;
65  }
66 
67  return preflight2(ctx);
68 }

71 NFTokenMint::createNFTokenID(
72  std::uint16_t flags,
73  std::uint16_t fee,
74  AccountID const& issuer,
75  nft::Taxon taxon,
76  std::uint32_t tokenSeq)
77 {
78  // An issuer may issue several NFTs with the same taxon; to ensure that NFTs
79  // are spread across multiple pages we lightly mix the taxon up by using the
80  // sequence (which is not under the issuer's direct control) as the seed for
81  // a simple linear congruential generator. cipheredTaxon() does this work.
82  taxon = nft::cipheredTaxon(tokenSeq, taxon);
83 
84  // The values are packed inside a 32-byte buffer, so we need to make sure
85  // that the endianess is fixed.
86  flags = boost::endian::native_to_big(flags);
87  fee = boost::endian::native_to_big(fee);
88  taxon = nft::toTaxon(boost::endian::native_to_big(nft::toUInt32(taxon)));
89  tokenSeq = boost::endian::native_to_big(tokenSeq);
90 
91  std::array<std::uint8_t, 32> buf{};
92 
93  auto ptr = buf.data();
94 
95  // This code is awkward but the idea is to pack these values into a single
96  // 256-bit value that uniquely identifies this NFT.
97  std::memcpy(ptr, &flags, sizeof(flags));
98  ptr += sizeof(flags);
99 
100  std::memcpy(ptr, &fee, sizeof(fee));
101  ptr += sizeof(fee);
102 
103  std::memcpy(ptr, issuer.data(), issuer.size());
104  ptr += issuer.size();
105 
106  std::memcpy(ptr, &taxon, sizeof(taxon));
107  ptr += sizeof(taxon);
108 
109  std::memcpy(ptr, &tokenSeq, sizeof(tokenSeq));
110  ptr += sizeof(tokenSeq);
111  assert(std::distance(buf.data(), ptr) == buf.size());
112 
113  return uint256::fromVoid(buf.data());
114 }

117 NFTokenMint::preclaim(PreclaimContext const& ctx)
118 {
119  // The issuer of the NFT may or may not be the account executing this
120  // transaction. Check that and verify that this is allowed:
121  if (auto issuer = ctx.tx[~sfIssuer])
122  {
123  auto const sle = ctx.view.read(keylet::account(*issuer));
124 
125  if (!sle)
126  return tecNO_ISSUER;
127 
128  if (auto const minter = (*sle)[~sfNFTokenMinter];
129  minter != ctx.tx[sfAccount])
130  return tecNO_PERMISSION;
131  }
132 
133  return tesSUCCESS;
134 }

136 TER
137 NFTokenMint::doApply()
138 {
139  auto const issuer = ctx_.tx[~sfIssuer].value_or(account_);
140 
141  auto const tokenSeq = [this, &issuer]() -> Expected<std::uint32_t, TER> {
142  auto const root = view().peek(keylet::account(issuer));
143  if (root == nullptr)
144  // Should not happen. Checked in preclaim.
145  return Unexpected(tecNO_ISSUER);
146 
147  // Get the unique sequence number for this token:
148  std::uint32_t const tokenSeq = (*root)[~sfMintedNFTokens].value_or(0);
149  {
150  std::uint32_t const nextTokenSeq = tokenSeq + 1;
151  if (nextTokenSeq < tokenSeq)
152  return Unexpected(tecMAX_SEQUENCE_REACHED);
153 
154  (*root)[sfMintedNFTokens] = nextTokenSeq;
155  }
156  ctx_.view().update(root);
157  return tokenSeq;
158  }();
159 
160  if (!tokenSeq.has_value())
161  return (tokenSeq.error());
162 
163  std::uint32_t const ownerCountBefore =
164  view().read(keylet::account(account_))->getFieldU32(sfOwnerCount);
165 
166  // Assemble the new NFToken.
167  SOTemplate const* nfTokenTemplate =
168  InnerObjectFormats::getInstance().findSOTemplateBySField(sfNFToken);
169 
170  if (nfTokenTemplate == nullptr)
171  // Should never happen.
172  return tecINTERNAL;
173 
174  STObject newToken(
175  *nfTokenTemplate,
176  sfNFToken,
177  [this, &issuer, &tokenSeq](STObject& object) {
178  object.setFieldH256(
179  sfNFTokenID,
180  createNFTokenID(
181  static_cast<std::uint16_t>(ctx_.tx.getFlags() & 0x0000FFFF),
182  ctx_.tx[~sfTransferFee].value_or(0),
183  issuer,
184  nft::toTaxon(ctx_.tx[sfNFTokenTaxon]),
185  tokenSeq.value()));
186 
187  if (auto const uri = ctx_.tx[~sfURI])
188  object.setFieldVL(sfURI, *uri);
189  });
190 
191  if (TER const ret =
192  nft::insertToken(ctx_.view(), account_, std::move(newToken));
193  ret != tesSUCCESS)
194  return ret;
195 
196  // Only check the reserve if the owner count actually changed. This
197  // allows NFTs to be added to the page (and burn fees) without
198  // requiring the reserve to be met each time. The reserve is
199  // only managed when a new NFT page is added.
200  if (auto const ownerCountAfter =
201  view().read(keylet::account(account_))->getFieldU32(sfOwnerCount);
202  ownerCountAfter > ownerCountBefore)
203  {
204  if (auto const reserve = view().fees().accountReserve(ownerCountAfter);
205  mPriorBalance < reserve)
206  return tecINSUFFICIENT_RESERVE;
207  }
208  return tesSUCCESS;
209 }
