86 template <class ConstBufferSequence>
87 std::size_t
88 write_some(ConstBufferSequence const& buffers, error_code& ec)
89 {
90 ec.assign(0, ec.category());
91 using boost::asio::buffer_cast;
92 using boost::asio::buffer_size;
93 auto const n = (std::min)(buffer_size(buffers), write_max_);
94 str.reserve(str.size() + n);
95 for (boost::asio::const_buffer buffer : buffer_prefix(n, buffers))
96 str.append(buffer_cast<char const*>(buffer), buffer_size(buffer));
97 return n;
98 }

107 handler};
108 get_io_service().post(
