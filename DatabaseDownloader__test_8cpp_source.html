49 struct DownloadCompleter
50 {
51 std::mutex m;
52 std::condition_variable cv;
53 bool called = false;
54 boost::filesystem::path dest;
55
56 void
57 operator()(boost::filesystem::path dst)
58 {
59 std::unique_lock<std::mutex> lk(m);
60 called = true;
61 dest = std::move(dst);
62 cv.notify_one();
63 };
64
65 bool
66 waitComplete()
67 {
68 std::unique_lock<std::mutex> lk(m);
69
70 auto stat = cv.wait_for(
71 lk, std::chrono::seconds(10), [this] { return called; });
72
73 called = false;
74 return stat;
75 };
76 };

127 void
128 testDownload(bool verify)
129 {
130 testcase << std::string("Basic download - SSL ") +
131 (verify ? "Verify" : "No Verify");
132
133 using namespace jtx;
134
135 ripple::test::detail::FileDirGuard cert{
136 *this, "_cert", "ca.pem", TrustedPublisherServer::ca_cert()};
137
138 Env env{*this, envconfig([&cert, &verify](std::unique_ptr<Config> cfg) {
139 if ((cfg->SSL_VERIFY = verify)) // yes, this is assignment
140 cfg->SSL_VERIFY_FILE = cert.file().string();
141 return cfg;
142 })};
143
144 Downloader dl{env};
145
146 // create a TrustedPublisherServer as a simple HTTP
147 // server to request from. Use the /textfile endpoint
148 // to get a simple text file sent as response.
149 auto server = createServer(env);
150 log << "Downloading DB from " << server->local_endpoint() << std::endl;
151
152 ripple::test::detail::FileDirGuard const data{
153 *this, "downloads", "data", "", false, false};
154 // initiate the download and wait for the callback
155 // to be invoked
156 auto stat = dl->download(
157 server->local_endpoint().address().to_string(),
158 std::to_string(server->local_endpoint().port()),
159 "/textfile",
160 11,
161 data.file(),
162 std::function<void(boost::filesystem::path)>{std::ref(cb)});
163 if (!BEAST_EXPECT(stat))
164 {
165 REPORT_FAILURE(dl);
166 return;
167 }
168 if (!BEAST_EXPECT(cb.waitComplete()))
169 {
170 REPORT_FAILURE(dl);
171 return;
172 }
173 BEAST_EXPECT(cb.dest == data.file());
174 if (!BEAST_EXPECT(boost::filesystem::exists(data.file())))
175 return;
176 BEAST_EXPECT(boost::filesystem::file_size(data.file()) > 0);
177 }

179 void
180 testFailures()
181 {
182 testcase("Error conditions");
183
184 using namespace jtx;
185
186 Env env{*this};
187
188 {
189 // bad hostname
190 boost::system::error_code ec;
191 boost::asio::ip::tcp::resolver resolver{env.app().getIOService()};
192 auto const results = resolver.resolve("badhostname", "443", ec);
193 // we require an error in resolving this name in order
194 // for this test to pass. Some networks might have DNS hijacking
195 // that prevent NXDOMAIN, in which case the failure is not
196 // possible, so we skip the test.
197 if (ec)
198 {
199 Downloader dl{env};
200 ripple::test::detail::FileDirGuard const datafile{
201 *this, "downloads", "data", "", false, false};
202 BEAST_EXPECT(dl->download(
203 "badhostname",
204 "443",
205 "",
206 11,
207 datafile.file(),
208 std::function<void(boost::filesystem::path)>{
209 std::ref(cb)}));
210 if (!BEAST_EXPECT(cb.waitComplete()))
211 {
212 REPORT_FAILURE(dl);
213 }
214 BEAST_EXPECT(!boost::filesystem::exists(datafile.file()));
215 BEAST_EXPECTS(
216 dl.sink_.messages().str().find("async_resolve") !=
217 std::string::npos,
218 dl.sink_.messages().str());
219 }
220 }
