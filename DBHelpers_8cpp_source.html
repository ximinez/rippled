26 static bool
27 writeToLedgersDB(LedgerInfo const& info, PgQuery& pgQuery, beast::Journal& j)
28 {
29  JLOG(j.debug()) << __func__;
30  auto cmd = boost::format(
31  R"(INSERT INTO ledgers
32  VALUES (%u,'\x%s', '\x%s',%u,%u,%u,%u,%u,'\x%s','\x%s'))");
33 
34  auto ledgerInsert = boost::str(
35  cmd % info.seq % strHex(info.hash) % strHex(info.parentHash) %
36  info.drops.drops() % info.closeTime.time_since_epoch().count() %
37  info.parentCloseTime.time_since_epoch().count() %
38  info.closeTimeResolution.count() % info.closeFlags %
39  strHex(info.accountHash) % strHex(info.txHash));
40  JLOG(j.trace()) << __func__ << " : "
41  << " : "
42  << "query string = " << ledgerInsert;
43 
44  auto res = pgQuery(ledgerInsert.data());
45 
46  return res;
47 }

49 bool
50 writeToPostgres(
51  LedgerInfo const& info,
52  std::vector<AccountTransactionsData> const& accountTxData,
53  std::shared_ptr<PgPool> const& pgPool,
54  beast::Journal& j)
55 {
56  JLOG(j.debug()) << __func__ << " : "
57  << "Beginning write to Postgres";
58 
59  try
60  {
61  // Create a PgQuery object to run multiple commands over the same
62  // connection in a single transaction block.
63  PgQuery pg(pgPool);
64  auto res = pg("BEGIN");
65  if (!res || res.status() != PGRES_COMMAND_OK)
66  {
67  std::stringstream msg;
68  msg << "bulkWriteToTable : Postgres insert error: " << res.msg();
69  Throw<std::runtime_error>(msg.str());
70  }
71 
72  // Writing to the ledgers db fails if the ledger already exists in the
73  // db. In this situation, the ETL process has detected there is another
74  // writer, and falls back to only publishing
75  if (!writeToLedgersDB(info, pg, j))
76  {
77  JLOG(j.warn()) << __func__ << " : "
78  << "Failed to write to ledgers database.";
79  return false;
80  }
81 
82  std::stringstream transactionsCopyBuffer;
83  std::stringstream accountTransactionsCopyBuffer;
84  for (auto const& data : accountTxData)
85  {
86  std::string txHash = strHex(data.txHash);
87  std::string nodestoreHash = strHex(data.nodestoreHash);
88  auto idx = data.transactionIndex;
89  auto ledgerSeq = data.ledgerSequence;
90 
91  transactionsCopyBuffer << std::to_string(ledgerSeq) << '\t'
92  << std::to_string(idx) << '\t' << "\\\\x"
93  << txHash << '\t' << "\\\\x" << nodestoreHash
94  << '\n';
95 
96  for (auto const& a : data.accounts)
97  {
98  std::string acct = strHex(a);
99  accountTransactionsCopyBuffer
100  << "\\\\x" << acct << '\t' << std::to_string(ledgerSeq)
101  << '\t' << std::to_string(idx) << '\n';
102  }
103  }
104 
105  pg.bulkInsert("transactions", transactionsCopyBuffer.str());
106  pg.bulkInsert(
107  "account_transactions", accountTransactionsCopyBuffer.str());
108 
109  res = pg("COMMIT");
110  if (!res || res.status() != PGRES_COMMAND_OK)
111  {
112  std::stringstream msg;
113  msg << "bulkWriteToTable : Postgres insert error: " << res.msg();
114  assert(false);
115  Throw<std::runtime_error>(msg.str());
116  }
117 
118  JLOG(j.info()) << __func__ << " : "
119  << "Successfully wrote to Postgres";
120  return true;
121  }
122  catch (std::exception& e)
123  {
124  JLOG(j.error()) << __func__ << "Caught exception writing to Postgres : "
125  << e.what();
126  assert(false);
127  return false;
128  }
129 }
