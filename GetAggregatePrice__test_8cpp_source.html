30class GetAggregatePrice_test : public beast::unit_test::suite
31{
32public:
33 void
34 testErrors()
35 {
36 testcase("Errors");
37 using namespace jtx;
38 using Oracles = std::vector<std::pair<Account, std::uint32_t>>;
39 Account const owner{"owner"};
40 Account const some{"some"};
41 static Oracles oracles = {{owner, 1}};
42
43 {
44 Env env(*this);
45 // missing base_asset
46 auto ret =
47 Oracle::aggregatePrice(env, std::nullopt, "USD", oracles);
48 BEAST_EXPECT(
49 ret[jss::error_message].asString() ==
50 "Missing field 'base_asset'.");
51
52 // missing quote_asset
53 ret = Oracle::aggregatePrice(env, "XRP", std::nullopt, oracles);
54 BEAST_EXPECT(
55 ret[jss::error_message].asString() ==
56 "Missing field 'quote_asset'.");
57
58 // missing oracles array
59 ret = Oracle::aggregatePrice(env, "XRP", "USD");
60 BEAST_EXPECT(
61 ret[jss::error_message].asString() ==
62 "Missing field 'oracles'.");
63
64 // empty oracles array
65 ret = Oracle::aggregatePrice(env, "XRP", "USD", Oracles{});
66 BEAST_EXPECT(ret[jss::error].asString() == "oracleMalformed");
67
68 // invalid oracle document id
69 ret = Oracle::aggregatePrice(env, "XRP", "USD", {{{owner, 2}}});
70 BEAST_EXPECT(ret[jss::error].asString() == "objectNotFound");
71
72 // invalid owner
73 ret = Oracle::aggregatePrice(env, "XRP", "USD", {{{some, 1}}});
74 BEAST_EXPECT(ret[jss::error].asString() == "objectNotFound");
75
76 // oracles have wrong asset pair
77 env.fund(XRP(1'000), owner);
78 Oracle oracle(
79 env, {.owner = owner, .series = {{"XRP", "EUR", 740, 1}}});
80 ret = Oracle::aggregatePrice(
81 env, "XRP", "USD", {{{owner, oracle.documentID()}}});
82 BEAST_EXPECT(ret[jss::error].asString() == "objectNotFound");
83
84 // invalid trim value
85 ret = Oracle::aggregatePrice(
86 env, "XRP", "USD", {{{owner, oracle.documentID()}}}, 0);
87 BEAST_EXPECT(ret[jss::error].asString() == "invalidParams");
88 ret = Oracle::aggregatePrice(
89 env, "XRP", "USD", {{{owner, oracle.documentID()}}}, 26);
90 BEAST_EXPECT(ret[jss::error].asString() == "invalidParams");
91 }
92
93 // too many oracles
94 {
95 Env env(*this);
96 std::vector<std::pair<Account, std::uint32_t>> oracles;
97 for (int i = 0; i < 201; ++i)
98 {
99 Account const owner(std::to_string(i));
100 env.fund(XRP(1'000), owner);
101 Oracle oracle(env, {.owner = owner, .documentID = i});
102 oracles.emplace_back(owner, oracle.documentID());
103 }
104 auto const ret = Oracle::aggregatePrice(env, "XRP", "USD", oracles);
105 BEAST_EXPECT(ret[jss::error].asString() == "oracleMalformed");
106 }
107 }

109 void
110 testRpc()
111 {
112 testcase("RPC");
113 using namespace jtx;
114
115 auto prep = [&](Env& env, auto& oracles) {
116 oracles.reserve(10);
117 for (int i = 0; i < 10; ++i)
118 {
119 Account const owner{std::to_string(i)};
120 env.fund(XRP(1'000), owner);
121 Oracle oracle(
122 env,
123 {.owner = owner,
124 .documentID = rand(),
125 .series = {
126 {"XRP", "USD", 740 + i, 1}, {"XRP", "EUR", 740, 1}}});
127 oracles.emplace_back(owner, oracle.documentID());
128 }
129 };
130
131 // Aggregate data set includes all price oracle instances, no trimming
132 // or time threshold
133 {
134 Env env(*this);
135 std::vector<std::pair<Account, std::uint32_t>> oracles;
136 prep(env, oracles);
137 // entire and trimmed stats
138 auto ret = Oracle::aggregatePrice(env, "XRP", "USD", oracles);
139 BEAST_EXPECT(ret[jss::entire_set][jss::mean] == "74.45");
140 BEAST_EXPECT(ret[jss::entire_set][jss::size].asUInt() == 10);
141 BEAST_EXPECT(
142 ret[jss::entire_set][jss::standard_deviation] ==
143 "0.3027650354097492");
144 BEAST_EXPECT(ret[jss::median] == "74.45");
145 BEAST_EXPECT(ret[jss::time] == 946694900);
146 }
147
148 // Aggregate data set includes all price oracle instances
149 {
150 Env env(*this);
151 std::vector<std::pair<Account, std::uint32_t>> oracles;
152 prep(env, oracles);
153 // entire and trimmed stats
154 auto ret =
155 Oracle::aggregatePrice(env, "XRP", "USD", oracles, 20, 100);
156 BEAST_EXPECT(ret[jss::entire_set][jss::mean] == "74.45");
157 BEAST_EXPECT(ret[jss::entire_set][jss::size].asUInt() == 10);
158 BEAST_EXPECT(
159 ret[jss::entire_set][jss::standard_deviation] ==
160 "0.3027650354097492");
161 BEAST_EXPECT(ret[jss::median] == "74.45");
162 BEAST_EXPECT(ret[jss::trimmed_set][jss::mean] == "74.45");
163 BEAST_EXPECT(ret[jss::trimmed_set][jss::size].asUInt() == 6);
164 BEAST_EXPECT(
165 ret[jss::trimmed_set][jss::standard_deviation] ==
166 "0.187082869338697");
167 BEAST_EXPECT(ret[jss::time] == 946694900);
168 }
169
170 // A reduced dataset, as some price oracles have data beyond three
171 // updated ledgers
172 {
173 Env env(*this);
174 std::vector<std::pair<Account, std::uint32_t>> oracles;
175 prep(env, oracles);
176 for (int i = 0; i < 3; ++i)
177 {
178 Oracle oracle(
179 env,
180 {.owner = oracles[i].first,
181 .documentID = oracles[i].second},
182 false);
183 // push XRP/USD by more than three ledgers, so this price
184 // oracle is not included in the dataset
185 oracle.set(UpdateArg{.series = {{"XRP", "EUR", 740, 1}}});
186 oracle.set(UpdateArg{.series = {{"XRP", "EUR", 740, 1}}});
187 oracle.set(UpdateArg{.series = {{"XRP", "EUR", 740, 1}}});
188 }
189 for (int i = 3; i < 6; ++i)
190 {
191 Oracle oracle(
192 env,
193 {.owner = oracles[i].first,
194 .documentID = oracles[i].second},
195 false);
196 // push XRP/USD by two ledgers, so this price
197 // is included in the dataset
198 oracle.set(UpdateArg{.series = {{"XRP", "EUR", 740, 1}}});
199 oracle.set(UpdateArg{.series = {{"XRP", "EUR", 740, 1}}});
200 }
201
202 // entire and trimmed stats
203 auto ret =
204 Oracle::aggregatePrice(env, "XRP", "USD", oracles, 20, 200);
205 BEAST_EXPECT(ret[jss::entire_set][jss::mean] == "74.6");
206 BEAST_EXPECT(ret[jss::entire_set][jss::size].asUInt() == 7);
207 BEAST_EXPECT(
208 ret[jss::entire_set][jss::standard_deviation] ==
209 "0.2160246899469287");
210 BEAST_EXPECT(ret[jss::median] == "74.6");
211 BEAST_EXPECT(ret[jss::trimmed_set][jss::mean] == "74.6");
212 BEAST_EXPECT(ret[jss::trimmed_set][jss::size].asUInt() == 5);
213 BEAST_EXPECT(
214 ret[jss::trimmed_set][jss::standard_deviation] ==
215 "0.158113883008419");
216 BEAST_EXPECT(ret[jss::time] == 946694900);
217 }
218
219 // Reduced data set because of the time threshold
220 {
221 Env env(*this);
222 std::vector<std::pair<Account, std::uint32_t>> oracles;
223 prep(env, oracles);
224 for (int i = 0; i < oracles.size(); ++i)
225 {
226 Oracle oracle(
227 env,
228 {.owner = oracles[i].first,
229 .documentID = oracles[i].second},
230 false);
231 // push XRP/USD by two ledgers, so this price
232 // is included in the dataset
233 oracle.set(UpdateArg{.series = {{"XRP", "USD", 740, 1}}});
234 }
235
236 // entire stats only, limit lastUpdateTime to {200, 125}
237 auto ret = Oracle::aggregatePrice(
238 env, "XRP", "USD", oracles, std::nullopt, 75);
239 BEAST_EXPECT(ret[jss::entire_set][jss::mean] == "74");
240 BEAST_EXPECT(ret[jss::entire_set][jss::size].asUInt() == 8);
241 BEAST_EXPECT(ret[jss::entire_set][jss::standard_deviation] == "0");
242 BEAST_EXPECT(ret[jss::median] == "74");
243 BEAST_EXPECT(ret[jss::time] == 946695000);
244 }
245 }

248 run() override
249 {
250 testErrors();
251 testRpc();
252 }
