59 uint160
60 DeterministicShard::hash(uint256 const& lastHash) const
61 {
62  using beast::hash_append;
63  ripemd160_hasher h;
64 
65  hash_append(h, lastHash);
66  hash_append(h, index_);
67  hash_append(h, db_.firstLedgerSeq(index_));
68  hash_append(h, db_.lastLedgerSeq(index_));
69  hash_append(h, Shard::version);
70 
71  auto const result = static_cast<ripemd160_hasher::result_type>(h);
72  return uint160::fromVoid(result.data());
73 }

75 std::uint64_t
76 DeterministicShard::digest(int n) const
77 {
78  auto const data = hash_.data();
79 
80  if (n == 2)
81  { // Extract 32 bits:
82  return (static_cast<std::uint64_t>(data[19]) << 24) +
83  (static_cast<std::uint64_t>(data[18]) << 16) +
84  (static_cast<std::uint64_t>(data[17]) << 8) +
85  (static_cast<std::uint64_t>(data[16]));
86  }
87 
88  std::uint64_t ret = 0;
89 
90  if (n == 0 || n == 1)
91  { // Extract 64 bits
92  for (int i = n; i < 16; i += 2)
93  ret = (ret << 8) + data[i];
94  }
95 
96  return ret;
97 }

99 bool
100 DeterministicShard::init()
101 {
102  if (index_ < db_.earliestShardIndex())
103  {
104  JLOG(j_.error()) << "shard " << index_ << " is illegal";
105  return false;
106  }
107 
108  Config const& config{app_.config()};
109 
110  Section section{config.section(ConfigSection::shardDatabase())};
111  std::string const type{get<std::string>(section, "type", "nudb")};
112 
113  if (type != "nudb")
114  {
115  JLOG(j_.error()) << "shard " << index_ << " backend type " << type
116  << " not supported";
117  return false;
118  }
119 
120  auto factory{Manager::instance().find(type)};
121  if (!factory)
122  {
123  JLOG(j_.error()) << "shard " << index_
124  << " failed to create factory for backend type "
125  << type;
126  return false;
127  }
128 
129  ctx_->start();
130 
131  section.set("path", tempdir_.string());
132  backend_ = factory->createInstance(
133  NodeObject::keyBytes, section, scheduler_, *ctx_, j_);
134 
135  if (!backend_)
136  {
137  JLOG(j_.error()) << "shard " << index_
138  << " failed to create backend type " << type;
139  return false;
140  }
141 
142  // Open or create the NuDB key/value store
143  bool preexist = exists(tempdir_);
144  if (preexist)
145  {
146  remove_all(tempdir_);
147  preexist = false;
148  }
149 
150  backend_->open(
151  !preexist,
152  digest(2) | 0x5348524400000000ll, /* appType */
153  digest(0), /* uid */
154  digest(1) /* salt */
155  );
156 
157  inited_ = true;
158 
159  return true;
160 }
