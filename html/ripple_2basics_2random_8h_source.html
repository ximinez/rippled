66 inline
67 beast::xor_shift_engine&
68 default_prng ()
69 {
70  static
71  boost::thread_specific_ptr<beast::xor_shift_engine> engine;
72 
73  if (!engine.get())
74  {
75  std::random_device rng;
76 
77  std::uint64_t seed = rng();
78 
79  for (int i = 0; i < 6; ++i)
80  {
81  if (seed == 0)
82  seed = rng();
83 
84  seed ^= (seed << (7 - i)) * rng();
85  }
86 
87  engine.reset (new beast::xor_shift_engine (seed));
88  }
89 
90  return *engine;
91 }
