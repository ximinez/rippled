31 NFTokenAcceptOffer::preflight(PreflightContext const& ctx)
32 {
33  if (!ctx.rules.enabled(featureNonFungibleTokensV1))
34  return temDISABLED;
35 
36  if (auto const ret = preflight1(ctx); !isTesSuccess(ret))
37  return ret;
38 
39  if (ctx.tx.getFlags() & tfNFTokenAcceptOfferMask)
40  return temINVALID_FLAG;
41 
42  auto const bo = ctx.tx[~sfNFTokenBuyOffer];
43  auto const so = ctx.tx[~sfNFTokenSellOffer];
44 
45  // At least one of these MUST be specified
46  if (!bo && !so)
47  return temMALFORMED;
48 
49  // The `BrokerFee` field must not be present in direct mode but may be
50  // present and greater than zero in brokered mode.
51  if (auto const bf = ctx.tx[~sfNFTokenBrokerFee])
52  {
53  if (!bo || !so)
54  return temMALFORMED;
55 
56  if (*bf <= beast::zero)
57  return temMALFORMED;
58  }
59 
60  return preflight2(ctx);
61 }

64 NFTokenAcceptOffer::preclaim(PreclaimContext const& ctx)
65 {
66  auto const checkOffer = [&ctx](std::optional<uint256> id)
67  -> std::pair<std::shared_ptr<const SLE>, TER> {
68  if (id)
69  {
70  auto offerSLE = ctx.view.read(keylet::nftoffer(*id));
71 
72  if (!offerSLE)
73  return {nullptr, tecOBJECT_NOT_FOUND};
74 
75  if (hasExpired(ctx.view, (*offerSLE)[~sfExpiration]))
76  return {nullptr, tecEXPIRED};
77 
78  return {std::move(offerSLE), tesSUCCESS};
79  }
80  return {nullptr, tesSUCCESS};
81  };
82 
83  auto const [bo, err1] = checkOffer(ctx.tx[~sfNFTokenBuyOffer]);
84  if (!isTesSuccess(err1))
85  return err1;
86  auto const [so, err2] = checkOffer(ctx.tx[~sfNFTokenSellOffer]);
87  if (!isTesSuccess(err2))
88  return err2;
89 
90  if (bo && so)
91  {
92  // Brokered mode:
93  // The two offers being brokered must be for the same token:
94  if ((*bo)[sfNFTokenID] != (*so)[sfNFTokenID])
95  return tecNFTOKEN_BUY_SELL_MISMATCH;
96 
97  // The two offers being brokered must be for the same asset:
98  if ((*bo)[sfAmount].issue() != (*so)[sfAmount].issue())
99  return tecNFTOKEN_BUY_SELL_MISMATCH;
100 
101  // Ensure that the buyer is willing to pay at least as much as the
102  // seller is requesting:
103  if ((*so)[sfAmount] > (*bo)[sfAmount])
104  return tecINSUFFICIENT_PAYMENT;
105 
106  // If the seller specified a destination, that destination must be
107  // the buyer or the broker.
108  if (auto const dest = so->at(~sfDestination))
109  {
110  if (*dest != bo->at(sfOwner) && *dest != ctx.tx[sfAccount])
111  return tecNFTOKEN_BUY_SELL_MISMATCH;
112  }
113 
114  // The broker can specify an amount that represents their cut; if they
115  // have, ensure that the seller will get at least as much as they want
116  // to get *after* this fee is accounted for (but before the issuer's
117  // cut, if any).
118  if (auto const brokerFee = ctx.tx[~sfNFTokenBrokerFee])
119  {
120  if (brokerFee->issue() != (*bo)[sfAmount].issue())
121  return tecNFTOKEN_BUY_SELL_MISMATCH;
122 
123  if (brokerFee >= (*bo)[sfAmount])
124  return tecINSUFFICIENT_PAYMENT;
125 
126  if ((*so)[sfAmount] > (*bo)[sfAmount] - *brokerFee)
127  return tecINSUFFICIENT_PAYMENT;
128  }
129  }
130 
131  if (bo)
132  {
133  if (((*bo)[sfFlags] & lsfSellNFToken) == lsfSellNFToken)
134  return tecNFTOKEN_OFFER_TYPE_MISMATCH;
135 
136  // An account can't accept an offer it placed:
137  if ((*bo)[sfOwner] == ctx.tx[sfAccount])
138  return tecCANT_ACCEPT_OWN_NFTOKEN_OFFER;
139 
140  // If not in bridged mode, the account must own the token:
141  if (!so &&
142  !nft::findToken(ctx.view, ctx.tx[sfAccount], (*bo)[sfNFTokenID]))
143  return tecNO_PERMISSION;
144 
145  // The account offering to buy must have funds:
146  auto const needed = bo->at(sfAmount);
147 
148  if (accountHolds(
149  ctx.view,
150  (*bo)[sfOwner],
151  needed.getCurrency(),
152  needed.getIssuer(),
153  fhZERO_IF_FROZEN,
154  ctx.j) < needed)
155  return tecINSUFFICIENT_FUNDS;
156  }
157 
158  if (so)
159  {
160  if (((*so)[sfFlags] & lsfSellNFToken) != lsfSellNFToken)
161  return tecNFTOKEN_OFFER_TYPE_MISMATCH;
162 
163  // An account can't accept an offer it placed:
164  if ((*so)[sfOwner] == ctx.tx[sfAccount])
165  return tecCANT_ACCEPT_OWN_NFTOKEN_OFFER;
166 
167  // The seller must own the token.
168  if (!nft::findToken(ctx.view, (*so)[sfOwner], (*so)[sfNFTokenID]))
169  return tecNO_PERMISSION;
170 
171  // If not in bridged mode...
172  if (!bo)
173  {
174  // If the offer has a Destination field, the acceptor must be the
175  // Destination.
176  if (auto const dest = so->at(~sfDestination);
177  dest.has_value() && *dest != ctx.tx[sfAccount])
178  return tecNO_PERMISSION;
179  }
180 
181  // The account offering to buy must have funds:
182  auto const needed = so->at(sfAmount);
183 
184  if (accountHolds(
185  ctx.view,
186  ctx.tx[sfAccount],
187  needed.getCurrency(),
188  needed.getIssuer(),
189  fhZERO_IF_FROZEN,
190  ctx.j) < needed)
191  return tecINSUFFICIENT_FUNDS;
192  }
193 
194  return tesSUCCESS;
195 }

211 NFTokenAcceptOffer::acceptOffer(std::shared_ptr<SLE> const& offer)
212 {
213  bool const isSell = offer->isFlag(lsfSellNFToken);
214  AccountID const owner = (*offer)[sfOwner];
215  AccountID const& seller = isSell ? owner : account_;
216  AccountID const& buyer = isSell ? account_ : owner;
217 
218  auto const nftokenID = (*offer)[sfNFTokenID];
219 
220  if (auto amount = offer->getFieldAmount(sfAmount); amount != beast::zero)
221  {
222  // Calculate the issuer's cut from this sale, if any:
223  if (auto const fee = nft::getTransferFee(nftokenID); fee != 0)
224  {
225  auto const cut = multiply(amount, nft::transferFeeAsRate(fee));
226 
227  if (auto const issuer = nft::getIssuer(nftokenID);
228  cut != beast::zero && seller != issuer && buyer != issuer)
229  {
230  if (auto const r = pay(buyer, issuer, cut); !isTesSuccess(r))
231  return r;
232  amount -= cut;
233  }
234  }
235 
236  // Send the remaining funds to the seller of the NFT
237  if (auto const r = pay(buyer, seller, amount); !isTesSuccess(r))
238  return r;
239  }
240 
241  // Now transfer the NFT:
242  auto tokenAndPage = nft::findTokenAndPage(view(), seller, nftokenID);
243 
244  if (!tokenAndPage)
245  return tecINTERNAL;
246 
247  if (auto const ret = nft::removeToken(
248  view(), seller, nftokenID, std::move(tokenAndPage->page));
249  !isTesSuccess(ret))
250  return ret;
251 
252  return nft::insertToken(view(), buyer, std::move(tokenAndPage->token));
253 }

256 NFTokenAcceptOffer::doApply()
257 {
258  auto const loadToken = [this](std::optional<uint256> const& id) {
259  std::shared_ptr<SLE> sle;
260  if (id)
261  sle = view().peek(keylet::nftoffer(*id));
262  return sle;
263  };
264 
265  auto bo = loadToken(ctx_.tx[~sfNFTokenBuyOffer]);
266  auto so = loadToken(ctx_.tx[~sfNFTokenSellOffer]);
267 
268  if (bo && !nft::deleteTokenOffer(view(), bo))
269  {
270  JLOG(j_.fatal()) << "Unable to delete buy offer '"
271  << to_string(bo->key()) << "': ignoring";
272  return tecINTERNAL;
273  }
274 
275  if (so && !nft::deleteTokenOffer(view(), so))
276  {
277  JLOG(j_.fatal()) << "Unable to delete sell offer '"
278  << to_string(so->key()) << "': ignoring";
279  return tecINTERNAL;
280  }
281 
282  // Bridging two different offers
283  if (bo && so)
284  {
285  AccountID const buyer = (*bo)[sfOwner];
286  AccountID const seller = (*so)[sfOwner];
287 
288  auto const nftokenID = (*so)[sfNFTokenID];
289 
290  // The amount is what the buyer of the NFT pays:
291  STAmount amount = (*bo)[sfAmount];
292 
293  // Three different folks may be paid. The order of operations is
294  // important.
295  //
296  // o The broker is paid the cut they requested.
297  // o The issuer's cut is calculated from what remains after the
298  // broker is paid. The issuer can take up to 50% of the remainder.
299  // o Finally, the seller gets whatever is left.
300  //
301  // It is important that the issuer's cut be calculated after the
302  // broker's portion is already removed. Calculating the issuer's
303  // cut before the broker's cut is removed can result in more money
304  // being paid out than the seller authorized. That would be bad!
305 
306  // Send the broker the amount they requested.
307  if (auto const cut = ctx_.tx[~sfNFTokenBrokerFee];
308  cut && cut.value() != beast::zero)
309  {
310  if (auto const r = pay(buyer, account_, cut.value());
311  !isTesSuccess(r))
312  return r;
313 
314  amount -= cut.value();
315  }
316 
317  // Calculate the issuer's cut, if any.
318  if (auto const fee = nft::getTransferFee(nftokenID);
319  amount != beast::zero && fee != 0)
320  {
321  auto cut = multiply(amount, nft::transferFeeAsRate(fee));
322 
323  if (auto const issuer = nft::getIssuer(nftokenID);
324  seller != issuer && buyer != issuer)
325  {
326  if (auto const r = pay(buyer, issuer, cut); !isTesSuccess(r))
327  return r;
328 
329  amount -= cut;
330  }
331  }
332 
333  // And send whatever remains to the seller.
334  if (amount > beast::zero)
335  {
336  if (auto const r = pay(buyer, seller, amount); !isTesSuccess(r))
337  return r;
338  }
339 
340  auto tokenAndPage = nft::findTokenAndPage(view(), seller, nftokenID);
341 
342  if (!tokenAndPage)
343  return tecINTERNAL;
344 
345  if (auto const ret = nft::removeToken(
346  view(), seller, nftokenID, std::move(tokenAndPage->page));
347  !isTesSuccess(ret))
348  return ret;
349 
350  return nft::insertToken(view(), buyer, std::move(tokenAndPage->token));
351  }
352 
353  if (bo)
354  return acceptOffer(bo);
355 
356  if (so)
357  return acceptOffer(so);
358 
359  return tecINTERNAL;
360 }
