105 bool
106 STValidation::isValid() const noexcept
107 {
108  if (!valid_)
109  {
110  assert(publicKeyType(getSignerPublic()) == KeyType::secp256k1);
111 
112  valid_ = verifyDigest(
113  getSignerPublic(),
114  getSigningHash(),
115  makeSlice(getFieldVL(sfSignature)),
116  getFlags() & vfFullyCanonicalSig);
117  }
118 
119  return valid_.value();
120 }
