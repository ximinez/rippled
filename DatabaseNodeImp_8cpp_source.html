46 DatabaseNodeImp::fetchNodeObject(
47  uint256 const& hash,
48  std::uint32_t,
49  FetchReport& fetchReport)
50 {
51  std::shared_ptr<NodeObject> nodeObject;
52  Status status;
53 
54  try
55  {
56  status = backend_->fetch(hash.data(), &nodeObject);
57  }
58  catch (std::exception const& e)
59  {
60  JLOG(j_.fatal()) << "Exception, " << e.what();
61  Rethrow();
62  }
63 
64  switch (status)
65  {
66  case ok:
67  ++fetchHitCount_;
68  if (nodeObject)
69  fetchSz_ += nodeObject->getData().size();
70  break;
71  case notFound:
72  break;
73  case dataCorrupt:
74  JLOG(j_.fatal()) << "Corrupt NodeObject #" << hash;
75  break;
76  default:
77  JLOG(j_.warn()) << "Unknown status=" << status;
78  break;
79  }
80 
81  if (nodeObject)
82  fetchReport.wasFound = true;
83 
84  return nodeObject;
85 }
