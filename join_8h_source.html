26 template <class Stream, class Iter>
27 Stream&
28 join(Stream& s, Iter iter, Iter end, std::string const& delimiter)
29 {
30  if (iter == end)
31  return s;
32  s << *iter;
33  for (++iter; iter != end; ++iter)
34  s << delimiter << *iter;
35  return s;
36 }
