61template <typename TIn, typename TOut>
62void
63AMMOffer<TIn, TOut>::consume(
64 ApplyView& view,
65 TAmounts<TIn, TOut> const& consumed)
66{
67 // Consumed offer must be less or equal to the original
68 if (consumed.in > amounts_.in || consumed.out > amounts_.out)
69 Throw<std::logic_error>("Invalid consumed AMM offer.");
70 // AMM pool is updated when the amounts are transferred
71 // in BookStep::consumeOffer().
72
73 consumed_ = true;
74
75 // Let the context know AMM offer is consumed
76 ammLiquidity_.context().setAMMUsed();
77}

79template <typename TIn, typename TOut>
80TAmounts<TIn, TOut>
81AMMOffer<TIn, TOut>::limitOut(
82 TAmounts<TIn, TOut> const& offrAmt,
83 TOut const& limit,
84 bool fixReducedOffers,
85 bool roundUp) const
86{
87 // Change the offer size proportionally to the original offer quality
88 // to keep the strands quality order unchanged. The taker pays slightly
89 // more for the offer in this case, which results in a slightly higher
90 // pool product than the original pool product. I.e. if the original
91 // pool is poolPays, poolGets and the offer is assetIn, assetOut then
92 // poolPays * poolGets < (poolPays - assetOut) * (poolGets + assetIn)
93 if (ammLiquidity_.multiPath())
94 {
95 if (fixReducedOffers)
96 // It turns out that the ceil_out implementation has some slop in
97 // it. ceil_out_strict removes that slop. But removing that slop
98 // affects transaction outcomes, so the change must be made using
99 // an amendment.
100 return quality().ceil_out_strict(offrAmt, limit, roundUp);
101 return quality().ceil_out(offrAmt, limit);
102 }
103 // Change the offer size according to the conservation function. The offer
104 // quality is increased in this case, but it doesn't matter since there is
105 // only one path.
106 return {swapAssetOut(balances_, limit, ammLiquidity_.tradingFee()), limit};
107}

110TAmounts<TIn, TOut>
111AMMOffer<TIn, TOut>::limitIn(
112 TAmounts<TIn, TOut> const& offrAmt,
113 TIn const& limit) const
114{
115 // See the comments above in limitOut().
116 if (ammLiquidity_.multiPath())
117 return quality().ceil_in(offrAmt, limit);
118 return {limit, swapAssetIn(balances_, limit, ammLiquidity_.tradingFee())};

131template <typename TIn, typename TOut>
132bool
133AMMOffer<TIn, TOut>::checkInvariant(
134 TAmounts<TIn, TOut> const& consumed,
135 beast::Journal j) const
136{
137 if (consumed.in > amounts_.in || consumed.out > amounts_.out)
138 {
139 JLOG(j.error()) << "AMMOffer::checkInvariant failed: consumed "
140 << to_string(consumed.in) << " "
141 << to_string(consumed.out) << " amounts "
142 << to_string(amounts_.in) << " "
143 << to_string(amounts_.out);
144
145 return false;
146 }
147
148 Number const product = balances_.in * balances_.out;
149 auto const newBalances = TAmounts<TIn, TOut>{
150 balances_.in + consumed.in, balances_.out - consumed.out};
151 Number const newProduct = newBalances.in * newBalances.out;
152
153 if (newProduct >= product ||
154 withinRelativeDistance(product, newProduct, Number{1, -7}))
155 return true;
156
157 JLOG(j.error()) << "AMMOffer::checkInvariant failed: balances "
158 << to_string(balances_.in) << " "
159 << to_string(balances_.out) << " new balances "
160 << to_string(newBalances.in) << " "
161 << to_string(newBalances.out) << " product/newProduct "
162 << product << " " << newProduct << " diff "
163 << (product != Number{0}
164 ? to_string((product - newProduct) / product)
165 : "undefined");
166 return false;
167}
