115 template <class ConstBufferSequence>
116 std::size_t
117 write_some(ConstBufferSequence const& buffers, error_code& ec)
118 {
119 ec.assign(0, ec.category());
120 using boost::asio::buffer_cast;
121 using boost::asio::buffer_size;
122 auto const n = buffer_size(buffers);
123 str.reserve(str.size() + n);
124 for (boost::asio::const_buffer buffer : buffers)
125 str.append(buffer_cast<char const*>(buffer), buffer_size(buffer));
126 return n;
127 }
